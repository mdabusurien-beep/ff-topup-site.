<script>

// üî• ‡¶è‡¶ñ‡¶æ‡¶®‡ßá ‡¶§‡ßã‡¶Æ‡¶æ‡¶∞ Firebase Config ‡¶¨‡¶∏‡¶æ‡¶¨‡ßá
var firebaseConfig = {
apiKey: "YOUR_API_KEY",
authDomain: "YOUR_AUTH_DOMAIN",
databaseURL: "YOUR_DATABASE_URL",
projectId: "YOUR_PROJECT_ID",
storageBucket: "YOUR_STORAGE_BUCKET",
messagingSenderId: "YOUR_MSG_ID",
appId: "YOUR_APP_ID"
};

firebase.initializeApp(firebaseConfig);
var database = firebase.database();

var adminPassword = "12345"; // change this

function placeOrder(){
var uid = document.getElementById("uid").value;
var pack = document.getElementById("package").value;
var number = document.getElementById("number").value;

if(uid==""||pack==""||number==""){
alert("Fill all fields");
return;
}

database.ref("orders").push({
uid:uid,
pack:pack,
number:number
});

alert("Order Submitted!");
}

function adminLogin(){
document.getElementById("adminPanel").style.display="block";
}

function login(){
var pass = document.getElementById("adminPass").value;
if(pass==adminPassword){
document.getElementById("ordersSection").style.display="block";
loadOrders();
}else{
alert("Wrong Password");
}
}

function loadOrders(){
database.ref("orders").on("value",function(snapshot){
var list=document.getElementById("ordersList");
list.innerHTML="";
snapshot.forEach(function(child){
var data=child.val();
var key=child.key;
list.innerHTML+=`
<div class="order">
UID: ${data.uid}<br>
Package: ${data.pack}<br>
Number: ${data.number}<br>
<button onclick="deleteOrder('${key}')">Delete</button>
</div>`;
});
});
}

function deleteOrder(key){
database.ref("orders/"+key).remove();
}
